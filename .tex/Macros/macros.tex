%	Name
\newcommand{\name}[1]{
	% Font Family - Helvetica 
	% \fontfamily{phv} \selectfont	
	% Print the name centered and bold
	\begin{center}
		{\Huge{\textbf{#1}}}
	\end{center}
	% Back to normal size and normal font
	% \normalfont	
}

%	Quality	
\newcommand{\quality}[1]{
	% \fontfamily{phv} \selectfont
	\begin{center}
		{\LARGE{\textsf{#1}}}
		% {\textsl{#1}}
	\end{center}
	% \normalfont
}

	% for text

}

%	Education
\newcommand{\education}[4]{
	\noindent
	% \hangindent=2em\hangafter=0 
	% Name of the studies
	\textbf{#1} \\
	% New paragraph with school in italics
	\textit{#2} \hfill
	\framebox{\parbox{6em}{\centering{#3}}} \par
	% \vspace*{0.5em}
	% NO hanging and in smaller text
	\noindent \hangindent=6em \hangafter=0 
	% \begin{center}

	% \end{center}
	 \par
}

	\noindent
